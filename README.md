## Install

```bash
npm install --save context-simplifier
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'context-simplifier'
import 'context-simplifier/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
